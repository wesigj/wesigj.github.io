class Counter {
private:
    int count;
public:
    Counter() : count(0) {}

    // é‡è½½å‰ç½®é€’å¢è¿ç®—ç¬¦
    Counter& operator++() {
        ++count;
        return *this;
    }

    // é‡è½½åç½®é€’å¢è¿ç®—ç¬¦
    Counter operator++(int) {
        Counter temp = *this;
        ++(*this);

    }
};

int main() {
    Counter c;
    std::cout << "Before increment: ";
    c.display();

    ++c;
    std::cout << "After prefix increment: ";
    c.display();

    c++;
    std::cout << "After postfix increment: ";
    c.display();
    return 0;
}
