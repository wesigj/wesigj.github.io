template <typename T>
void merge(T *begin, T *mid, T *end, T *temp) {
    int i = 0, j = 0, k = 0;
    while (begin + i < mid && mid + j < end) {
        if (*(begin + i) <= *(mid + j)) {
            temp[k++] = *(begin + i++);
        } else {
            temp[k++] = *(mid + j++);
        }
    }
    while (begin + i < mid) {
        temp[k++] = *(begin + i++);
    }
    while (mid + j < end) {
        temp[k++] = *(mid + j++);
    }
    std::copy(temp, temp + k, begin);
}
